[438. 找到字符串中所有字母异位词](https://leetcode.cn/problems/find-all-anagrams-in-a-string/)

### 思路

这个和找一个异位词是的思路是一样的，只是判断的时候需要判断到结尾

### 题解

```c++
class Solution {
private:
    bool isAllZero(vector<int>& path) {
        for(auto& iter : path) {
            if(iter != 0) {
                return false;
            }
        }
        return true;
    }
public:
    vector<int> findAnagrams(string s, string p) {
        int n1 = s.size();
        int n2 = p.size();
        
        if(n1 < n2) {
            return {};
        }
        
        vector<int> path(26, 0);
        vector<int> ans;
        for(int i = 0; i < n2; ++i) {
            path[s[i] - 'a']--;
            path[p[i] - 'a']++;
        }
        
        if(isAllZero(path)) {
            ans.emplace_back(0);
        }
        
        for(int i = n2; i < n1; ++i) {
            path[s[i - n2] - 'a']++;
            path[s[i] - 'a']--;
            if(isAllZero(path)) {
                ans.emplace_back(i - n2 + 1);
            }
        }
        
        retur
    }
};
```
